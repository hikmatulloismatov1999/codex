\section{Definition and statement}

Throughout, $m$ denotes Lebesgue measure on $\mathbb{R}^n$, and all functions
are Lebesgue measurable. For $A\subset\mathbb{R}^n$ and
$y\in\mathbb{R}^n$, write
\[
    A-y := \{x\in\mathbb{R}^n : x+y\in A\}.
\]
The set $A-y$ is measurable whenever $A$ is measurable, and
$m(A-y)=m(A)$ by translation invariance of Lebesgue measure.

\begin{definition}[Almost equicontinuity]\label{def:ae}
Let $\mathcal{F}$ be a family of real-valued measurable functions on
$\mathbb{R}^n$. We say that $\mathcal{F}$ is \emph{almost equicontinuous} if
\[
    \forall \varepsilon>0\;\; \exists \delta>0\;\;
    \forall f\in\mathcal{F}\;\;
    \exists B_f\subset\mathbb{R}^n \text{ measurable}
\]
such that
\[
    m(B_f)<\varepsilon,
\]
and
\[
    x_1,x_2\in\mathbb{R}^n\setminus B_f,\quad |x_1-x_2|<\delta
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\varepsilon.
\]
\end{definition}

\begin{remark}[Equivalent two-tolerance formulation]\label{rem:two-params}
Definition~\ref{def:ae} is equivalent to the following formulation: for every
$\eta>0$ and every $\gamma>0$, there exists $\rho>0$ such that for every
$f\in\mathcal{F}$ there is a measurable set $B_f\subset\mathbb{R}^n$ with
$m(B_f)<\gamma$ and
\[
    x_1,x_2\in\mathbb{R}^n\setminus B_f,\quad |x_1-x_2|<\rho
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eta.
\]
Indeed, the one-parameter definition implies this statement by applying
Definition~\ref{def:ae} with
\[
    \varepsilon:=\min(\eta,\gamma).
\]
Then $m(B_f)<\varepsilon\le \gamma$ and the oscillation is bounded by
$\varepsilon\le \eta$. Conversely, taking $\eta=\gamma=\varepsilon$ gives the
one-parameter definition.

In the proof below we use the one-parameter definition directly, choosing the
single parameter small enough to control both the measure of $B_f$ and the
oscillation of $f$ on $\mathbb{R}^n\setminus B_f$.
\end{remark}

\begin{theorem}\label{thm:main}
Let $E\subset\mathbb{R}^n$ be a measurable set with $m(E)<\infty$, and let
$\mathcal{F}$ be a family of real-valued measurable functions on
$\mathbb{R}^n$ such that:
\begin{enumerate}
    \item[\textup{(H1)}] $f(x)=0$ for every $x\notin E$ and every
    $f\in\mathcal{F}$;
    \item[\textup{(H2)}] $\mathcal{F}$ is almost equicontinuous in the sense of
    Definition~\ref{def:ae}.
\end{enumerate}
Then
\[
    \lim_{|y|\to 0}\;\sup_{f\in\mathcal{F}}
    \int_{\mathbb{R}^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx = 0.
\]
Equivalently, for every $\varepsilon^{*}>0$ there exists $r>0$ such that for
every $|y|<r$,
\[
    \sup_{f\in\mathcal{F}}
    \int_{\mathbb{R}^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    < \varepsilon^{*}.
\]
\end{theorem}

\section{Proof}

\begin{proof}
For $f\in\mathcal{F}$ and $y\in\mathbb{R}^n$, set
\[
    I(f,y):=
    \int_{\mathbb{R}^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx.
\]
We prove that $I(f,y)$ is small uniformly in $f$ when $|y|$ is small.

\medskip
\noindent\textbf{The null-support case.}
Suppose first that $m(E)=0$. By (H1), $\{f\neq 0\}\subset E$ for every
$f\in\mathcal{F}$, so $f=0$ almost everywhere. By translation invariance,
$f(\cdot+y)=0$ almost everywhere as well. Hence $I(f,y)=0$ for every
$f\in\mathcal{F}$ and every $y\in\mathbb{R}^n$. Thus the conclusion holds with
any $r>0$. We now assume $m(E)>0$.

\medskip
\noindent\textbf{Choice of parameters.}
Fix $\varepsilon^{*}>0$. We use $\varepsilon^{*}$ for the target accuracy in
the conclusion and reserve $\varepsilon$ for the parameter in
Definition~\ref{def:ae}. Choose
\begin{equation}\label{eq:choice-eps}
    \varepsilon :=
    \min\left\{\frac{\varepsilon^{*}}{8m(E)},\frac{\varepsilon^{*}}{8}\right\}.
\end{equation}
This number is positive because $0<m(E)<\infty$. By
Definition~\ref{def:ae}, there exists $\delta>0$ such that for every
$f\in\mathcal{F}$ there is a measurable set $B_f\subset\mathbb{R}^n$ with
\begin{equation}\label{eq:Bf-bound}
    m(B_f)<\varepsilon
\end{equation}
and
\begin{equation}\label{eq:ae-implication}
    x_1,x_2\in\mathbb{R}^n\setminus B_f,\quad |x_1-x_2|<\delta
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\varepsilon.
\end{equation}
Set $r:=\delta$. We will show that
\[
    \sup_{f\in\mathcal{F}} I(f,y)<\varepsilon^{*}
    \quad\text{whenever } |y|<r.
\]

Fix $f\in\mathcal{F}$ and $y\in\mathbb{R}^n$ with $|y|<r=\delta$. The case
$y=0$ is immediate, so the following estimates also cover all nonzero
$y$ with $|y|<r$.

\medskip
\noindent\textbf{Localization of the integrand.}
The integrand vanishes outside $E\cup(E-y)$. Indeed, if
$x\notin E\cup(E-y)$, then $x\notin E$ and $x+y\notin E$. By (H1),
$f(x)=0$ and $f(x+y)=0$. Therefore
\begin{equation}\label{eq:localized}
    I(f,y)=
    \int_{E\cup(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx.
\end{equation}
Moreover,
\begin{equation}\label{eq:E-size}
    m(E\cup(E-y))\le m(E)+m(E-y)=2m(E).
\end{equation}

\medskip
\noindent\textbf{Good and bad regions.}
Define
\[
    \operatorname{Bad}(f,y):=B_f\cup(B_f-y),
    \qquad
    G(f,y):=(E\cup(E-y))\setminus \operatorname{Bad}(f,y).
\]
By translation invariance and subadditivity,
\begin{equation}\label{eq:bad-size}
    m(\operatorname{Bad}(f,y))
    \le m(B_f)+m(B_f-y)
    =2m(B_f)
    <2\varepsilon.
\end{equation}

If $x\in G(f,y)$, then $x\notin B_f$ and $x\notin B_f-y$, hence
$x+y\notin B_f$. Since $|x-(x+y)|=|y|<\delta$, \eqref{eq:ae-implication}
gives
\begin{equation}\label{eq:pointwise}
    |f(x+y)-f(x)|<\varepsilon
    \qquad (x\in G(f,y)).
\end{equation}

\medskip
\noindent\textbf{Splitting the integral.}
Using \eqref{eq:localized}, split
\[
    I(f,y)=I_{\mathrm{good}}+I_{\mathrm{bad}},
\]
where
\[
    I_{\mathrm{good}}:=
    \int_{G(f,y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
\]
and
\[
    I_{\mathrm{bad}}:=
    \int_{(E\cup(E-y))\cap\operatorname{Bad}(f,y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx.
\]
On the good region, \eqref{eq:pointwise} and $\min(a,1)\le a$ imply
\[
\begin{aligned}
    I_{\mathrm{good}}
    &\le \varepsilon\,m(G(f,y)) \\
    &\le \varepsilon\,m(E\cup(E-y)) \\
    &\le 2\varepsilon\,m(E) \\
    &\le \frac{\varepsilon^{*}}{4},
\end{aligned}
\]
where the last inequality follows from \eqref{eq:choice-eps}. On the bad
region, the truncation gives $\min(a,1)\le 1$, and hence
\[
\begin{aligned}
    I_{\mathrm{bad}}
    &\le m\bigl((E\cup(E-y))\cap\operatorname{Bad}(f,y)\bigr) \\
    &\le m(\operatorname{Bad}(f,y)) \\
    &<2\varepsilon \\
    &\le \frac{\varepsilon^{*}}{4}.
\end{aligned}
\]
Thus
\[
    I(f,y)<\frac{\varepsilon^{*}}{2}
\]
for every $f\in\mathcal{F}$ and every $|y|<r$. Taking the supremum over
$f\in\mathcal{F}$ gives
\[
    \sup_{f\in\mathcal{F}} I(f,y)
    \le \frac{\varepsilon^{*}}{2}
    <\varepsilon^{*}
    \qquad (|y|<r).
\]
Since $\varepsilon^{*}>0$ was arbitrary, the desired limit follows.
\end{proof}

\section{Remarks}

\begin{remark}
The proof uses only the standard one-parameter definition of almost
equicontinuity. The choice
\[
    \varepsilon =
    \min\left\{\frac{\varepsilon^{*}}{8m(E)},\frac{\varepsilon^{*}}{8}\right\}
\]
is simply a bookkeeping device: the same small parameter controls both the
measure of the exceptional set and the oscillation on its complement.
\end{remark}

\begin{remark}
The truncation $\min(\cdot,1)$ is essential for this argument. On the bad
region we do not assume uniform boundedness or an $L^p$ bound on
$\mathcal{F}$; the estimate is possible because the integrand is bounded by
$1$.
\end{remark}

\begin{remark}
The assumption $m(E)<\infty$ is used to control the good-region contribution:
\[
    I_{\mathrm{good}}\le
    \varepsilon\,m(E\cup(E-y))\le 2\varepsilon\,m(E).
\]
Without finite-measure support, this estimate gives no useful bound.
\end{remark}
