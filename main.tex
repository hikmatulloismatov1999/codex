\section{Definition and statement}

Throughout, $m$ denotes Lebesgue measure on $\mathbb{R}^n$, and all functions
are Lebesgue measurable. If $A\subset \mathbb{R}^n$ and
$y\in\mathbb{R}^n$, we write
\[
    A-y := \{x\in\mathbb{R}^n : x+y\in A\}.
\]
Thus $A-y$ is measurable whenever $A$ is measurable, and
$m(A-y)=m(A)$ by translation invariance of Lebesgue measure.

\begin{definition}[Classical almost equicontinuity]\label{def:ae}
Let $\mathcal{F}$ be a family of real-valued measurable functions on
$\mathbb{R}^n$. We say that $\mathcal{F}$ is \emph{almost equicontinuous} if
for every $\gamma>0$ there exists a measurable set
$B_\gamma\subset\mathbb{R}^n$ with
\[
    m(B_\gamma)<\gamma
\]
such that $\mathcal{F}$ is equicontinuous on
$\mathbb{R}^n\setminus B_\gamma$. Equivalently, for every $\eta>0$ there
exists $\rho>0$ such that
\[
    x_1,x_2\in \mathbb{R}^n\setminus B_\gamma,\quad |x_1-x_2|<\rho
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eta
\]
for every $f\in\mathcal{F}$.
\end{definition}

\begin{theorem}\label{thm:main}
Let $E\subset\mathbb{R}^n$ be measurable with $m(E)<\infty$, and let
$\mathcal{F}$ be a family of real-valued measurable functions on
$\mathbb{R}^n$. Assume that:
\begin{enumerate}
    \item[\textup{(H1)}] $f(x)=0$ for every $x\notin E$ and every
    $f\in\mathcal{F}$;
    \item[\textup{(H2)}] $\mathcal{F}$ is almost equicontinuous in the sense of
    Definition~\ref{def:ae}.
\end{enumerate}
Then
\[
    \lim_{|y|\to 0}\;\sup_{f\in\mathcal{F}}
    \int_{\mathbb{R}^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx = 0.
\]
Equivalently, for every $\varepsilon>0$ there exists $r>0$ such that, whenever
$|y|<r$,
\[
    \sup_{f\in\mathcal{F}}
    \int_{\mathbb{R}^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    <\varepsilon.
\]
\end{theorem}

\section{Proof}

\begin{proof}
For $f\in\mathcal{F}$ and $y\in\mathbb{R}^n$, set
\[
    I(f,y):=
    \int_{\mathbb{R}^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx .
\]
We prove that $I(f,y)$ is small uniformly in $f$ when $|y|$ is small.

If $m(E)=0$, then by (H1) each $f\in\mathcal{F}$ is zero almost everywhere.
By translation invariance, $f(\cdot+y)$ is also zero almost everywhere, and
hence $I(f,y)=0$ for all $f$ and $y$. Thus the conclusion is immediate. We may
therefore assume that $0<m(E)<\infty$.

Fix $\varepsilon>0$. Choose
\[
    \gamma:=\frac{\varepsilon}{8},
    \qquad
    \eta:=\frac{\varepsilon}{8\,m(E)}.
\]
By almost equicontinuity, there exists a measurable set
$B:=B_\gamma\subset\mathbb{R}^n$ such that
\[
    m(B)<\gamma
\]
and such that $\mathcal{F}$ is equicontinuous on
$\mathbb{R}^n\setminus B$. Applying this equicontinuity with the above
$\eta$, we obtain $\rho>0$ such that, for all $f\in\mathcal{F}$,
\[
    x_1,x_2\in\mathbb{R}^n\setminus B,\quad |x_1-x_2|<\rho
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eta .
\]
Set $r:=\rho$, and fix $y\in\mathbb{R}^n$ with $|y|<r$.

For every $f\in\mathcal{F}$, the integrand in $I(f,y)$ vanishes outside
$E\cup(E-y)$. Indeed, if $x\notin E\cup(E-y)$, then $x\notin E$ and
$x+y\notin E$, so (H1) gives $f(x)=f(x+y)=0$. Hence
\[
    I(f,y)=
    \int_{E\cup(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx .
\]
Moreover,
\[
    m(E\cup(E-y))\le m(E)+m(E-y)=2m(E).
\]

Define the bad set associated with the translation $y$ by
\[
    \operatorname{Bad}(y):=B\cup(B-y),
\]
and define the good set by
\[
    G(y):=(E\cup(E-y))\setminus \operatorname{Bad}(y).
\]
Since $m(B-y)=m(B)$, we have
\[
    m(\operatorname{Bad}(y))
    \le m(B)+m(B-y)
    =2m(B)
    <2\gamma .
\]

If $x\in G(y)$, then $x\notin B$ and $x\notin B-y$, hence
$x+y\notin B$. Since $|x-(x+y)|=|y|<\rho$, the equicontinuity estimate gives
\[
    |f(x+y)-f(x)|<\eta
\]
for every $f\in\mathcal{F}$. Therefore
\[
\begin{aligned}
    \int_{G(y)}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    &\le \eta\,m(G(y)) \\
    &\le \eta\,m(E\cup(E-y)) \\
    &\le 2\eta\,m(E)
     = \frac{\varepsilon}{4}.
\end{aligned}
\]
On the bad part, we use only the truncation bound $\min(a,1)\le 1$:
\[
\begin{aligned}
    \int_{(E\cup(E-y))\cap\operatorname{Bad}(y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    &\le m(\operatorname{Bad}(y))  \\
    &<2\gamma
     =\frac{\varepsilon}{4}.
\end{aligned}
\]
Combining the good and bad estimates, we get
\[
    I(f,y)<\frac{\varepsilon}{2}
\]
for every $f\in\mathcal{F}$ and every $|y|<r$. Taking the supremum over
$f\in\mathcal{F}$ yields
\[
    \sup_{f\in\mathcal{F}} I(f,y)
    \le \frac{\varepsilon}{2}
    <\varepsilon
    \qquad (|y|<r).
\]
Since $\varepsilon>0$ was arbitrary, the desired translation condition follows.
\end{proof}

\section{Remarks}

\begin{remark}
The proof uses the classical definition of almost equicontinuity: after removing
one set of arbitrarily small measure, the whole family is equicontinuous on the
remaining set. The numbers $\gamma$ and $\eta$ in the proof have different
roles: $\gamma$ controls the size of the exceptional set, while $\eta$ is the
ordinary tolerance in the equicontinuity estimate. This is not a new definition,
only an unpacking of the usual one.
\end{remark}

\begin{remark}
The truncation $\min(\cdot,1)$ is essential for this argument. On the bad set we
do not assume any uniform $L^p$ bound or uniform boundedness of the functions;
the estimate is possible only because the integrand is bounded by $1$.
\end{remark}

\begin{remark}
The assumption $m(E)<\infty$ is used to control the good part of the integral by
\[
    \eta\,m(E\cup(E-y))\le 2\eta\,m(E).
\]
Without finite measure support, this estimate gives no information.
\end{remark}
