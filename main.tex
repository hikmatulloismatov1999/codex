\author{Hikmatullo Ismatov}

\begin{abstract}
We prove that a family of measurable functions supported in a fixed
finite-measure set satisfies the truncated translation condition whenever it is
almost equicontinuous. We also show by a bounded smooth example on $\R$ that
the finite-measure support hypothesis cannot simply be removed, and we record
a replacement result under a uniform tail-support condition. The proof
separates the finite-measure estimate into three
contributions: a symmetric-difference part controlled by translation
continuity of finite-measure sets, a regular part controlled by almost
equicontinuity, and an exceptional part controlled by the truncation.
\end{abstract}

\section{Introduction}

Translation conditions are central in the classical
Kolmogorov--Riesz--Fréchet compactness theory; see, for instance, the modern
discussion in~\cite{HancheOlsenHolden}. In the asymptotic $L^p$ framework,
the ordinary $L^p$ distance is replaced by a truncated quantity involving
$\min(|f|,1)$, and the corresponding compactness theory has recently been
developed by Alves~\cite{AlvesKR,AlvesF}. Related comparisons between
asymptotic $L^p$ convergence and classical modes of convergence are discussed
in~\cite{AlvesOniani}.

The purpose of this note is to isolate several elementary facts in this circle
of ideas. First, we show in Theorem~\ref{thm:main} that almost
equicontinuity, together with support in a fixed set of finite measure,
implies the truncated translation condition. The proof is included in detail
because the truncation and the finite-measure support assumption play distinct
roles.

Second, Example~\ref{ex:sin} shows that the finite-measure assumption is not
cosmetic: even a singleton family consisting of one bounded smooth function on
$\R$ may fail the truncated translation condition. Third,
Theorem~\ref{thm:tail} proves that the same conclusion remains valid on
infinite-measure domains if the supports of the family are uniformly tight in
measure. Finally, after two examples related to the converse direction, we
prove in Theorem~\ref{thm:converse} that the interior truncated translation
condition implies almost equicontinuity on a finite-measure domain.

\section{Definitions and statement}

Throughout, $\meas{A}$ denotes the Lebesgue measure of a measurable set
$A\subset\R^n$, and all functions are Lebesgue measurable. For
$A\subset\R^n$ and $y\in\R^n$, we write
\[
    A-y:=\{x\in\R^n:x+y\in A\}.
\]
By translation invariance of Lebesgue measure, $A-y$ is measurable and
$\meas{A-y}=\meas{A}$. We also write
\[
    A\triangle B := (A\setminus B)\cup(B\setminus A)
\]
for the symmetric difference of two sets.

\begin{definition}[Almost equicontinuity]\label{def:ae}
Let $\F$ be a family of real-valued measurable functions on $\R^n$. We say
that $\F$ is \emph{almost equicontinuous} if for every $\eps>0$ there exists
$\delta>0$ such that, for every $f\in\F$, there is a measurable set
$B_f\subset\R^n$ satisfying
\[
    \meas{B_f}<\eps,
\]
and such that
\[
    x_1,x_2\in\R^n\setminus B_f,\quad |x_1-x_2|<\delta
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eps .
\]
\end{definition}

\begin{definition}[Almost equicontinuity on a set]\label{def:ae-domain}
Let $E\subset\R^n$ be measurable, and let $\F$ be a family of real-valued
measurable functions on $E$. We say that $\F$ is \emph{almost
equicontinuous on $E$} if for every $\eps>0$ there exists $\delta>0$ such
that, for every $f\in\F$, there is a measurable set $B_f\subset E$
satisfying
\[
    \meas{B_f}<\eps,
\]
and such that
\[
    x_1,x_2\in E\setminus B_f,\quad |x_1-x_2|<\delta
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eps .
\]
\end{definition}

\begin{theorem}\label{thm:main}
Let $E\subset\R^n$ be measurable with $\meas{E}<\infty$, and let $\F$ be a
family of real-valued measurable functions on $\R^n$. Assume that:
\begin{enumerate}[label=\textup{(H\arabic*)},ref=\textup{(H\arabic*)}]
    \item\label{hyp:support} $f(x)=0$ for every $x\notin E$ and every
    $f\in\F$;
    \item\label{hyp:ae} $\F$ is almost equicontinuous.
\end{enumerate}
Then
\[
    \lim_{|y|\to 0}\sup_{f\in\F}
    \int_{\R^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx=0.
\]
\end{theorem}

\begin{lemma}\label{lem:symdiff}
If $E\subset\R^n$ is measurable and $\meas{E}<\infty$, then
\[
    \lim_{|y|\to 0}\meas{E\triangle(E-y)}=0.
\]
\end{lemma}

\begin{proof}
Fix $\eta>0$. By regularity of Lebesgue measure, there exists a finite union
of bounded rectangles $Q\subset\R^n$ such that
\[
    \meas{E\triangle Q}<\frac{\eta}{3}.
\]
Since
\[
    E\triangle(E-y)
    \subset (E\triangle Q)\cup\bigl(Q\triangle(Q-y)\bigr)
    \cup\bigl((Q-y)\triangle(E-y)\bigr),
\]
translation invariance gives
\[
    \meas{E\triangle(E-y)}
    \le 2\meas{E\triangle Q}+\meas{Q\triangle(Q-y)}.
\]
It remains only to note that
\[
    \meas{Q\triangle(Q-y)}\longrightarrow 0
    \qquad (y\to0).
\]
Indeed, if $R=\prod_{j=1}^n(a_j,b_j)$ is a bounded rectangle, then
$R\triangle(R-y)$ is contained in a union of slabs whose total measure tends
to $0$ as $y\to0$. The same conclusion follows for a finite union of
rectangles by subadditivity. Therefore, for all sufficiently small $|y|$, we have
$\meas{Q\triangle(Q-y)}<\eta/3$, and consequently
\[
    \meas{E\triangle(E-y)}<\eta.
\]
\end{proof}

\section{Proof}

\begin{proof}
For $f\in\F$ and $y\in\R^n$, set
\[
    I(f,y):=
    \int_{\R^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx .
\]
We prove that $I(f,y)$ tends to zero as $y\to0$, uniformly with respect to
$f\in\F$.

If $\meas{E}=0$, then hypothesis~\ref{hyp:support} implies that every
$f\in\F$ is zero almost everywhere. By translation invariance,
$f(\cdot+y)$ is also zero almost everywhere for every $y\in\R^n$. Hence
$I(f,y)=0$ for all $f\in\F$ and all $y\in\R^n$. We therefore assume that
\[
    0<\meas{E}<\infty.
\]

Let $\widetilde\eps>0$ be given. Choose
\begin{equation}\label{eq:epsilon-choice}
    \eps:=
    \min\left\{
        \frac{\widetilde\eps}{3\meas{E}},
        \frac{\widetilde\eps}{6}
    \right\}.
\end{equation}
Lemma~\ref{lem:symdiff} gives $r_0>0$ such that
\begin{equation}\label{eq:symdiff-small}
    |y|<r_0
    \quad\Longrightarrow\quad
    \meas{E\triangle(E-y)}<\frac{\widetilde\eps}{3}.
\end{equation}
By hypothesis~\ref{hyp:ae} and Definition~\ref{def:ae}, there exists
$\delta>0$ such that for every $f\in\F$ there is a measurable set
$B_f\subset\R^n$ with
\begin{equation}\label{eq:exceptional-size}
    \meas{B_f}<\eps
\end{equation}
and
\begin{equation}\label{eq:ae-control}
    x_1,x_2\in\R^n\setminus B_f,\quad |x_1-x_2|<\delta
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eps .
\end{equation}
Set $r:=\min\{r_0,\delta\}$.

Fix $y\in\R^n$ with $|y|<r$, and fix $f\in\F$. If
$x\notin E\cup(E-y)$, then both $x$ and $x+y$ lie outside $E$. By
hypothesis~\ref{hyp:support}, $f(x)=f(x+y)=0$. Therefore the integrand in
$I(f,y)$ is supported in $E\cup(E-y)$, and so
\begin{equation}\label{eq:localization}
    I(f,y)
    =
    \int_{E\cup(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx .
\end{equation}

We split the localized integral into the symmetric-difference part and the
intersection part. Since the integrand is bounded by $1$, the
symmetric-difference estimate in equation~\eqref{eq:symdiff-small} gives
\begin{equation}\label{eq:symdiff-part}
    \int_{E\triangle(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    <\frac{\widetilde\eps}{3}.
\end{equation}

It remains to estimate the contribution over $E\cap(E-y)$. Put
\[
    \widetilde B:=B_f\cup(B_f-y).
\]
The exceptional-set estimate in equation~\eqref{eq:exceptional-size}, together
with translation invariance, gives
\begin{equation}\label{eq:Btilde-size}
    \meas{\widetilde B}
    \le \meas{B_f}+\meas{B_f-y}
    =2\meas{B_f}
    <2\eps .
\end{equation}

On $(E\cap(E-y))\setminus\widetilde B$, both $x$ and $x+y$ lie outside
$B_f$. Since $|x-(x+y)|=|y|<\delta$, the almost-equicontinuity estimate in
equation~\eqref{eq:ae-control} gives
\[
    |f(x+y)-f(x)|<\eps .
\]
Therefore
\begin{equation}\label{eq:regular-part}
    \int_{(E\cap(E-y))\setminus\widetilde B}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    \le \eps\,\meas{E}
    \le \frac{\widetilde\eps}{3}.
\end{equation}
On the exceptional part, the integrand is bounded by $1$, and the estimate in
equation~\eqref{eq:Btilde-size} gives
\begin{equation}\label{eq:exceptional-part}
    \int_{(E\cap(E-y))\cap\widetilde B}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    \le \meas{\widetilde B}
    <2\eps
    \le \frac{\widetilde\eps}{3}.
\end{equation}

Combining the localization identity~\eqref{eq:localization} with the preceding
three estimates, we obtain
\[
    I(f,y)<\widetilde\eps .
\]
This estimate is uniform in $f\in\F$, and therefore
\[
    \sup_{f\in\F}I(f,y)\le \widetilde\eps
    \qquad (|y|<r).
\]
Since $\widetilde\eps>0$ was arbitrary, the desired limit follows.
\end{proof}

\section{Necessity of the finite-measure hypothesis}

\begin{example}\label{ex:sin}
Let $E=\R$, and let $\F=\{f\}$, where
\[
    f(x):=\sin x .
\]
Since $E=\R$, the zero extension outside $E$ is simply the function itself.
The family $\F$ is almost equicontinuous, but it does not satisfy the
truncated translation condition in Theorem~\ref{thm:main}.
\end{example}

\begin{proof}
We first verify almost equicontinuity. The function $f$ is globally
Lipschitz, because
\[
    |\sin x-\sin z|\le |x-z|
    \qquad (x,z\in\R).
\]
Given $\eps>0$, choose $\delta=\eps$ and take the exceptional set
$B_f=\varnothing$. If $x,z\in\R$ and $|x-z|<\delta$, then
$|f(x)-f(z)|<\eps$. Thus Definition~\ref{def:ae} is satisfied with no
exceptional set, and the singleton family $\F$ is almost equicontinuous.

We now show that the truncated translation integral is infinite for every
small nonzero translation. Fix $y$ with $0<|y|<\pi/3$, and set
\[
    a_y:=2\left|\sin\frac{y}{2}\right|.
\]
Then $0<a_y<1$. The trigonometric identity
\[
    |\sin(x+y)-\sin x|
    =
    a_y\left|\cos\left(x+\frac{y}{2}\right)\right|
\]
therefore gives, for every $x\in\R$,
\[
    \min\bigl(|\sin(x+y)-\sin x|,1\bigr)
    =
    a_y\left|\cos\left(x+\frac{y}{2}\right)\right|,
\]
because the right-hand side is bounded above by $a_y<1$.

For $N\in\mathbb N$, integrate over the disjoint intervals
\[
    J_k:=\left[2\pi k-\frac{y}{2},\,2\pi(k+1)-\frac{y}{2}\right],
    \qquad k=0,\dots,N-1.
\]
Using the change of variables $u=x+y/2$, we obtain
\[
    \int_{\bigcup_{k=0}^{N-1}J_k}
    \min\bigl(|\sin(x+y)-\sin x|,1\bigr)\,dx
    =
    a_y\sum_{k=0}^{N-1}
    \int_{2\pi k}^{2\pi(k+1)}|\cos u|\,du .
\]
Since $\int_0^{2\pi}|\cos u|\,du=4$, the last expression equals
$4Na_y$. Letting $N\to\infty$ yields
\[
    \int_{\R}\min\bigl(|\sin(x+y)-\sin x|,1\bigr)\,dx=\infty
    \qquad (0<|y|<\pi/3).
\]
Since $\F$ contains only $f$, the supremum over $\F$ is the same infinite
quantity. Hence the limit in Theorem~\ref{thm:main} cannot hold when the
finite-measure support hypothesis is removed.
\end{proof}

\section{A uniform tail version}

\begin{theorem}\label{thm:tail}
Let $\F$ be a family of real-valued measurable functions on $\R^n$. Assume
that:
\begin{enumerate}[label=\textup{(T\arabic*)},ref=\textup{(T\arabic*)}]
    \item\label{tail:ae} $\F$ is almost equicontinuous;
    \item\label{tail:tight} the supports of the functions in $\F$ are
    uniformly tight in measure, in the sense that
    \[
        \lim_{R\to\infty}\sup_{f\in\F}
        \meas{\{x\in\R^n: |x|>R,\ f(x)\ne0\}}=0.
    \]
\end{enumerate}
Then
\[
    \lim_{|y|\to 0}\sup_{f\in\F}
    \int_{\R^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx=0.
\]
\end{theorem}

\begin{proof}
For $R>0$, write
\[
    B_R:=\{x\in\R^n: |x|\le R\},
    \qquad
    T_R(f):=\{x\in\R^n: |x|>R,\ f(x)\ne0\}.
\]
Fix $\widetilde\eps>0$. By hypothesis~\ref{tail:tight}, choose $R>0$ so
large that
\begin{equation}\label{eq:tail-small}
    \sup_{f\in\F}\meas{T_R(f)}<\frac{\widetilde\eps}{8}.
\end{equation}
Since $\meas{B_R}<\infty$, Lemma~\ref{lem:symdiff} gives $r_0>0$ such that
\begin{equation}\label{eq:ball-symdiff-small}
    |y|<r_0
    \quad\Longrightarrow\quad
    \meas{B_R\triangle(B_R-y)}<\frac{\widetilde\eps}{4}.
\end{equation}

Choose
\[
    \eps:=
    \min\left\{
        \frac{\widetilde\eps}{4\meas{B_R}},
        \frac{\widetilde\eps}{8}
    \right\}.
\]
By hypothesis~\ref{tail:ae} and Definition~\ref{def:ae}, there exists
$\delta>0$ such that for every $f\in\F$ there is a measurable set
$B_f\subset\R^n$ with
\begin{equation}\label{eq:tail-exceptional-small}
    \meas{B_f}<\eps
\end{equation}
and
\begin{equation}\label{eq:tail-ae-control}
    x_1,x_2\in\R^n\setminus B_f,\quad |x_1-x_2|<\delta
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eps .
\end{equation}
Set $r:=\min\{r_0,\delta\}$.

Fix $f\in\F$ and $y\in\R^n$ with $|y|<r$. We split the integral over
$\R^n$ into three regions:
\[
    B_R\cap(B_R-y),\qquad
    B_R\triangle(B_R-y),\qquad
    \R^n\setminus\bigl(B_R\cup(B_R-y)\bigr).
\]
The boundary region is controlled directly by
\eqref{eq:ball-symdiff-small}, since the integrand is bounded by $1$.

On the interior region $B_R\cap(B_R-y)$, put
\[
    \widetilde B:=B_f\cup(B_f-y).
\]
By \eqref{eq:tail-exceptional-small} and translation invariance,
\[
    \meas{\widetilde B}<2\eps .
\]
On $(B_R\cap(B_R-y))\setminus\widetilde B$, both $x$ and $x+y$ lie outside
$B_f$, and therefore \eqref{eq:tail-ae-control} gives
$|f(x+y)-f(x)|<\eps$. Hence
\begin{equation}\label{eq:tail-interior}
    \int_{B_R\cap(B_R-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    \le \eps\,\meas{B_R}+2\eps
    \le \frac{\widetilde\eps}{2}.
\end{equation}

It remains to estimate the exterior region. If
$x\notin B_R\cup(B_R-y)$ and the integrand is nonzero, then either
$x\in T_R(f)$ or $x+y\in T_R(f)$. Consequently,
\begin{equation}\label{eq:tail-exterior}
    \int_{\R^n\setminus(B_R\cup(B_R-y))}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    \le \meas{T_R(f)}+\meas{T_R(f)-y}
    <\frac{\widetilde\eps}{4},
\end{equation}
where we used \eqref{eq:tail-small} and translation invariance.

Combining \eqref{eq:ball-symdiff-small}, \eqref{eq:tail-interior}, and
\eqref{eq:tail-exterior}, we obtain
\[
    \int_{\R^n}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx
    <\widetilde\eps .
\]
The estimate is uniform in $f\in\F$. Since $\widetilde\eps>0$ was arbitrary,
the desired limit follows.
\end{proof}

\section{Examples for the converse direction}

The opposite direction asks whether the truncated translation condition can
force almost equicontinuity. The following two examples are useful tests for
any proposed converse theorem.

\begin{example}[moving intervals]\label{ex:moving-intervals}
Let
\[
    \F:=\{\mathbf 1_{(a,a+1)}:a\in\R\}.
\]
Then $\F$ satisfies the truncated translation condition and is almost
equicontinuous.
\end{example}

\begin{proof}
For $f_a:=\mathbf 1_{(a,a+1)}$, the integrand
$\min(|f_a(x+y)-f_a(x)|,1)$ is the indicator function of the symmetric
difference of $(a,a+1)$ and $(a-y,a+1-y)$. Hence, for $|y|<1$,
\[
    \int_{\R}\min\bigl(|f_a(x+y)-f_a(x)|,1\bigr)\,dx
    =
    \meas{(a,a+1)\triangle(a-y,a+1-y)}
    \le 2|y|.
\]
The estimate is independent of $a$, and therefore
\[
    \lim_{|y|\to0}\sup_{f\in\F}
    \int_{\R}\min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx=0.
\]

We now verify almost equicontinuity. Given $\eps>0$, choose
$\rho>0$ so small that $\rho<1/4$ and $4\rho<\eps$, and set
$\delta:=\rho$. For $f_a\in\F$, define
\[
    B_a:=(a-\rho,a+\rho)\cup(a+1-\rho,a+1+\rho).
\]
Then $\meas{B_a}\le4\rho<\eps$. If $x,z\in\R\setminus B_a$ and
$|x-z|<\delta$, then the two points cannot lie on opposite sides of either
endpoint $a$ or $a+1$. Thus $f_a(x)=f_a(z)$, and hence
$|f_a(x)-f_a(z)|=0<\eps$.
\end{proof}

\begin{example}[oscillating combs]\label{ex:combs}
For $n\ge1$, let
\[
    A_n:=
    \bigcup_{k=0}^{n-1}
    \left(\frac{k}{n},\,\frac{k}{n}+\frac{1}{2n}\right),
    \qquad
    f_n:=\mathbf 1_{A_n}.
\]
Then the family $\F:=\{f_n:n\ge1\}$ is not almost equicontinuous, and it
does not satisfy the truncated translation condition.
\end{example}

\begin{proof}
We first show that almost equicontinuity fails. Fix the accuracy
$\eps_0:=1/2$. Let $\delta>0$ be arbitrary, and choose $n$ so large that
$1/n<\delta$. In each interval
\[
    I_{n,k}:=\left(\frac{k}{n},\,\frac{k+1}{n}\right),
    \qquad k=0,\dots,n-1,
\]
the function $f_n$ is equal to $1$ on the left half and $0$ on the right half.
If a measurable set $B\subset\R$ satisfies the almost-equicontinuity
implication for $f_n$ at scale $\delta$ and accuracy $\eps_0$, then $B$ must
contain one of the two half-intervals in each $I_{n,k}$. Otherwise there
would be points $x,z\in I_{n,k}\setminus B$ in opposite
halves, and then $|x-z|<1/n<\delta$ while $|f_n(x)-f_n(z)|=1$.
Consequently,
\[
    \meas{B}\ge
    \sum_{k=0}^{n-1}\frac{1}{2n}
    =\frac12.
\]
This is impossible if the exceptional set is required to have measure less
than $\eps_0=1/2$. Therefore $\F$ is not almost equicontinuous.

The same oscillation also violates the truncated translation condition. Let
$y_n:=1/(2n)$. Then $y_n\to0$, and for every
$x\in(0,1-y_n)$, except for finitely many endpoints, the points $x$ and
$x+y_n$ lie in opposite halves of adjacent or identical subintervals. Hence
\[
    |f_n(x+y_n)-f_n(x)|=1
    \qquad\text{for a.e. }x\in(0,1-y_n).
\]
It follows that
\[
    \sup_{f\in\F}
    \int_{\R}\min\bigl(|f(x+y_n)-f(x)|,1\bigr)\,dx
    \ge
    \int_0^{1-y_n}1\,dx
    =1-y_n.
\]
Since $1-y_n\to1$, the truncated translation condition cannot hold.
\end{proof}

\section{A finite-measure converse on the domain}

We now prove a converse statement for functions defined on a common
finite-measure set. In this formulation only pairs of points that remain in
the domain are tested, so the natural translation condition is
\[
    \lim_{|y|\to0}\sup_{f\in\F}
    \int_{E\cap(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx=0.
\]
Here almost equicontinuity on $E$ is meant in the sense of
Definition~\ref{def:ae-domain}.

\begin{lemma}\label{lem:bad-measure-from-translation}
Let $E\subset\R^n$ be measurable, and let $\F$ be a family of real-valued
measurable functions on $E$. If
\[
    \lim_{|y|\to0}\sup_{f\in\F}
    \int_{E\cap(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx=0,
\]
then, for every $0<a\le1$,
\[
    \lim_{|y|\to0}\sup_{f\in\F}
    \meas{\{x\in E\cap(E-y): |f(x+y)-f(x)|\ge a\}}=0.
\]
\end{lemma}

\begin{proof}
On the set where $|f(x+y)-f(x)|\ge a$, the truncated integrand is at least
$a$. Hence
\[
    a\,\meas{\{x\in E\cap(E-y): |f(x+y)-f(x)|\ge a\}}
    \le
    \int_{E\cap(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx .
\]
Taking the supremum over $f\in\F$ and letting $y\to0$ proves the claim.
\end{proof}

\begin{lemma}[separated values]\label{lem:separated-values}
Let $A$ be a measurable set of finite measure, let $u:A\to\R$ be measurable,
and let $a>0$. Suppose that one must remove measure at least $\beta$ from
$A$ in order to make the remaining values of $u$ lie in an interval of length
$a$. Then
\[
    \meas{\{(x,z)\in A\times A: |u(x)-u(z)|\ge a\}}
    \ge \frac{\beta^2}{2}.
\]
\end{lemma}

\begin{proof}
If $\beta=0$, the assertion is trivial. Assume $\beta>0$. Let $\nu$ be the
push-forward of Lebesgue measure on $A$ under $u$. By hypothesis, every
closed interval in $\R$ of length $a$ misses $\nu$-measure at least $\beta$.
Choose
\[
    t:=\inf\{r\in\R:\nu((-\infty,r])\ge\beta/2\}.
\]
Then $\nu((-\infty,t])\ge\beta/2$ and
$\nu((-\infty,t))\le\beta/2$. If
$\nu((t+a,\infty))<\beta/2$, then
\[
    \nu(\R\setminus[t,t+a])
    =
    \nu((-\infty,t))+\nu((t+a,\infty))
    <\beta,
\]
which contradicts the preceding paragraph. Hence both separated tails
$(-\infty,t]$ and $(t+a,\infty)$ have $\nu$-measure at least $\beta/2$.
Pairs chosen from these two tails have values separated by at least $a$, and
therefore the bad set in $A\times A$ has measure at least
\[
    2\cdot\frac{\beta}{2}\cdot\frac{\beta}{2}
    =\frac{\beta^2}{2}.
\]
\end{proof}

\begin{lemma}[shifted cubes]\label{lem:shifted-cubes}
For each $s>0$ there are $3^n$ shifted grids of closed cubes of side length
$s$ with the following property: if $x,z\in\R^n$ and
$|x-z|<s/(3\sqrt n)$, then $x$ and $z$ belong to a common cube of one of
the grids.
\end{lemma}

\begin{proof}
In one dimension, the three grids
\[
    \{[ks+js/3,(k+1)s+js/3]:k\in\mathbb Z\},
    \qquad j=0,1,2,
\]
have the property that every interval of length less than $s/3$ is contained
in some interval from one of the three grids. Taking products of these three
one-dimensional grids gives $3^n$ grids in $\R^n$. If
$|x-z|<s/(3\sqrt n)$, then each coordinate difference is less than $s/3$,
and the one-dimensional assertion applies coordinate by coordinate.
\end{proof}

\begin{theorem}[finite-measure converse]\label{thm:converse}
Let $E\subset\R^n$ be measurable with $\meas{E}<\infty$, and let $\F$ be a
family of real-valued measurable functions on $E$. Assume that
\[
    \lim_{|y|\to0}\sup_{f\in\F}
    \int_{E\cap(E-y)}
    \min\bigl(|f(x+y)-f(x)|,1\bigr)\,dx=0.
\]
Then $\F$ is almost equicontinuous on $E$ in the sense of
Definition~\ref{def:ae-domain}.
\end{theorem}

\begin{proof}
We argue by contradiction. Suppose that $\F$ is not almost equicontinuous on
$E$. Then there exists an accuracy $0<\eps\le1$ such that, for every
$\delta>0$, one can find $f\in\F$ for which no measurable set
$B\subset E$ with $\meas{B}<\eps$ has the property
\[
    x,z\in E\setminus B,\quad |x-z|<\delta
    \quad\Longrightarrow\quad
    |f(x)-f(z)|<\eps .
\]

Choose a compact set $K\subset E$ such that
$\meas{E\setminus K}<\eps/4$. Fix $0<s\le1$, and let
$\mathcal G_1,\dots,\mathcal G_{3^n}$ be the shifted grids from
Lemma~\ref{lem:shifted-cubes}. Since $K$ is compact, the total number of
cubes from these grids that meet $K$ is at most $C_Ks^{-n}$, where $C_K$ is
independent of $s$.

Set $\delta_s:=s/(3\sqrt n)$, and choose $f_s\in\F$ witnessing the failure
of almost equicontinuity at scale $\delta_s$. For a cube $Q$, let
$\beta_Q$ be the infimum of $\meas{C}$ over all measurable
$C\subset K\cap Q$ such that the values of $f_s$ on
$(K\cap Q)\setminus C$ lie in some interval of length $\eps/2$.

The sum of these local quantities over all shifted grids is at least
$3\eps/4$. Indeed, if the sum were smaller, then after choosing local
exceptional sets whose measures are arbitrarily close to the corresponding
infima, the total local exceptional measure would still be below
$3\eps/4$. Adding $E\setminus K$, one would obtain a set $B\subset E$ with
$\meas{B}<\eps$.
Whenever $x,z\in E\setminus B$ and $|x-z|<\delta_s$,
Lemma~\ref{lem:shifted-cubes} puts $x$ and $z$ in a common cube of one
shifted grid. On that cube the remaining values of $f_s$ lie in an interval
of length $\eps/2$, so $|f_s(x)-f_s(z)|<\eps$, a contradiction. Thus
\begin{equation}\label{eq:beta-sum}
    \sum_{i=1}^{3^n}\sum_{Q\in\mathcal G_i}\beta_Q
    \ge \frac{3\eps}{4}.
\end{equation}

By Lemma~\ref{lem:separated-values}, applied on $K\cap Q$ with
$a=\eps/2$, we have
\[
    \meas{\{(x,z)\in(K\cap Q)^2:
    |f_s(x)-f_s(z)|\ge\eps/2\}}
    \ge \frac{\beta_Q^2}{2}.
\]
Using Cauchy's inequality, the counting bound for cubes meeting $K$, and
\eqref{eq:beta-sum}, we obtain a constant $c_1>0$, independent of $s$, such
that
\begin{equation}\label{eq:bad-pairs}
    \sum_{i=1}^{3^n}\sum_{Q\in\mathcal G_i}
    \meas{\{(x,z)\in(K\cap Q)^2:
    |f_s(x)-f_s(z)|\ge\eps/2\}}
    \ge c_1s^n .
\end{equation}

If two points lie in a common cube of side length $s$, then their distance is
at most $\sqrt n\,s$. Conversely, a fixed pair of points can belong to
$(K\cap Q)^2$ for only boundedly many cubes $Q$ among all the shifted grids;
the bound depends only on $n$. Thus, after absorbing this bounded overlap
into the constant, writing $z=x+y$, and using Fubini's theorem,
\eqref{eq:bad-pairs} implies
\[
    \int_{|y|\le\sqrt n\,s}
    \meas{\{x\in E\cap(E-y):
    |f_s(x+y)-f_s(x)|\ge\eps/2\}}\,dy
    \ge c_2s^n
\]
for some constant $c_2>0$ independent of $s$. Since the ball
$\{y:|y|\le\sqrt n\,s\}$ has measure comparable to $s^n$, there exists
$y_s\in\R^n$ with $|y_s|\le\sqrt n\,s$ such that
\begin{equation}\label{eq:bad-translation}
    \meas{\{x\in E\cap(E-y_s):
    |f_s(x+y_s)-f_s(x)|\ge\eps/2\}}\ge c_3
\end{equation}
for a constant $c_3>0$ independent of $s$.

Letting $s\downarrow0$, we have $y_s\to0$. By
Lemma~\ref{lem:bad-measure-from-translation}, the lower bound in
\eqref{eq:bad-translation} contradicts the assumed translation condition.
Therefore $\F$ must be almost equicontinuous on $E$.
\end{proof}

\begin{problem}\label{prob:converse}
Theorem~\ref{thm:converse} treats the interior translation condition on a
finite-measure domain. A remaining question is to identify the best analogue
for zero extensions on all of $\R^n$, especially when boundary effects and
infinite-measure tails are both present.
\end{problem}

\section{Remarks}

\begin{remark}
The finite-measure support assumption is used in two places. First, it implies
the symmetric-difference convergence in Lemma~\ref{lem:symdiff}. Second, it
controls the regular part of the intersection by
\[
    \eps\,\meas{E}.
\]
\end{remark}

\begin{remark}
The truncation $\min(\cdot,1)$ is used only on the symmetric-difference and
exceptional parts. No pointwise bound or uniform $L^p$ bound on the family is
needed for those estimates.
\end{remark}
