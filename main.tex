\author{Hikmatullo Ismatov}

\begin{abstract}
We compare classical almost equicontinuity for families of measurable
functions with the translation condition appearing in Kolmogorov--Riesz type
compactness theorems for asymptotic $L^p$ spaces. For families supported in a
fixed set of finite Lebesgue measure, we prove directly that almost
equicontinuity implies the truncated translation condition. Conversely, under
the natural almost equiboundedness hypothesis, the translation condition
implies almost equicontinuity as a consequence of the Kolmogorov--Riesz theorem
in asymptotic $L^p$ spaces. We also include examples clarifying why a common
exceptional set is too strong and why almost equiboundedness is independent of
translation control.
\end{abstract}

\section{Introduction}

The classical Kolmogorov--Riesz theorem characterizes relatively compact
subsets of $L^p(\R^n)$ by means of tightness and uniform translation
continuity. In the asymptotic $L^p$ setting, the relevant modular is
\[
    \phip(f):=\int_{\R^n}\min(|f(x)|,1)^p\,dx,
    \qquad 1\le p<\infty.
\]
The corresponding translation condition is
\[
    \lim_{|y|\to 0}\sup_{f\in\F}
    \phip(\tau_y f-f)=0,
    \qquad \tau_y f(x):=f(x+y).
\]
This condition measures continuity in the topology generated by the truncated
modular, rather than in the ordinary $L^p$ norm.

Recent work of Alves develops the functional analytic structure of these
spaces and proves a Kolmogorov--Riesz compactness theorem in asymptotic
$L^p(\R^n)$. In contrast with the classical $L^p$ theory, an additional
almost equiboundedness condition is necessary. The purpose of the present note
is narrower: we isolate the relation between almost equicontinuity and the
translation condition, especially on families supported in a set of finite
measure.

The first main observation is elementary: classical almost equicontinuity
implies the translation condition. The reverse implication is more delicate. In
the compactness setting, it follows from Alves's theorem together with the
classical compactness criterion for convergence in measure, provided one also
assumes almost equiboundedness. The examples at the end show why these
formulations should not be strengthened carelessly.

\section{Definitions}

Throughout, $m$ denotes Lebesgue measure on $\R^n$, and all functions are
Lebesgue measurable. If $A\subset \R^n$ and $y\in\R^n$, set
\[
    A-y:=\{x\in\R^n:x+y\in A\}.
\]
Then $A-y$ is measurable whenever $A$ is measurable, and
$m(A-y)=m(A)$.

\begin{definition}[Truncated translation condition]\label{def:tc}
Let $1\le p<\infty$. A family $\F$ satisfies the $p$-truncated translation
condition if
\[
    \lim_{|y|\to 0}\sup_{f\in\F}
    \int_{\R^n}\min(|f(x+y)-f(x)|,1)^p\,dx=0.
\]
\end{definition}

\begin{definition}[Classical almost equicontinuity]\label{def:ae}
A family $\F$ of real-valued measurable functions on $\R^n$ is called
almost equicontinuous if for every $\eps>0$ there exists $\rho>0$ such that
for every $f\in\F$ there is a measurable set $B_f\subset\R^n$ satisfying
$m(B_f)<\eps$ and
\[
    x_1,x_2\in\R^n\setminus B_f,\quad |x_1-x_2|<\rho
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\eps .
\]
\end{definition}

\begin{definition}[Almost equiboundedness]\label{def:aeb}
A family $\F$ is almost equibounded if for every $\eps>0$ there exists
$M>0$ such that
\[
    \sup_{f\in\F}m(\{x\in\R^n:|f(x)|>M\})<\eps .
\]
\end{definition}

\section{Almost equicontinuity implies translation control}

\begin{theorem}\label{thm:ae-to-tc}
Let $1\le p<\infty$. Let $E\subset\R^n$ be measurable with $m(E)<\infty$,
and let $\F$ be a family of measurable functions such that
\[
    f(x)=0 \quad \text{for every }x\notin E
    \text{ and every } f\in\F.
\]
If $\F$ is almost equicontinuous, then $\F$ satisfies the $p$-truncated
translation condition.
\end{theorem}

\begin{proof}
For $f\in\F$ and $y\in\R^n$, write
\[
    I_p(f,y):=
    \int_{\R^n}\min(|f(x+y)-f(x)|,1)^p\,dx.
\]
If $m(E)=0$, then every $f\in\F$ vanishes almost everywhere, and therefore
$I_p(f,y)=0$ for all $f$ and $y$. Hence we may assume $m(E)>0$.

Fix $\eps>0$. Choose $\alpha>0$ so small that
\begin{equation}\label{eq:alpha-choice}
    2m(E)\alpha^p+2\alpha<\eps .
\end{equation}
By almost equicontinuity, there exists $\rho>0$ such that for every
$f\in\F$ there is a measurable set $B_f$ with $m(B_f)<\alpha$ and
\[
    x_1,x_2\in\R^n\setminus B_f,\quad |x_1-x_2|<\rho
    \quad\Longrightarrow\quad
    |f(x_1)-f(x_2)|<\alpha .
\]
Let $|y|<\rho$ and fix $f\in\F$.

The integrand defining $I_p(f,y)$ vanishes outside $E\cup(E-y)$. Indeed, if
$x\notin E\cup(E-y)$, then $x\notin E$ and $x+y\notin E$, so
$f(x)=f(x+y)=0$. Hence
\[
    I_p(f,y)
    =
    \int_{E\cup(E-y)}\min(|f(x+y)-f(x)|,1)^p\,dx .
\]
Also
\[
    m(E\cup(E-y))\le m(E)+m(E-y)=2m(E).
\]

Define
\[
    \operatorname{Bad}(f,y):=B_f\cup(B_f-y),
    \qquad
    G(f,y):=(E\cup(E-y))\setminus \operatorname{Bad}(f,y).
\]
Then
\[
    m(\operatorname{Bad}(f,y))
    \le m(B_f)+m(B_f-y)
    =2m(B_f)<2\alpha.
\]
If $x\in G(f,y)$, then $x\notin B_f$ and $x+y\notin B_f$. Since
$|x-(x+y)|=|y|<\rho$, the almost equicontinuity estimate gives
\[
    |f(x+y)-f(x)|<\alpha .
\]
Therefore
\[
\begin{aligned}
    \int_{G(f,y)}\min(|f(x+y)-f(x)|,1)^p\,dx
    &\le \alpha^p\,m(G(f,y))  \\
    &\le 2m(E)\alpha^p .
\end{aligned}
\]
On the bad region we only use $\min(a,1)^p\le 1$:
\[
    \int_{(E\cup(E-y))\cap\operatorname{Bad}(f,y)}
    \min(|f(x+y)-f(x)|,1)^p\,dx
    \le m(\operatorname{Bad}(f,y))<2\alpha .
\]
Combining these two estimates and using \eqref{eq:alpha-choice}, we get
\[
    I_p(f,y)<\eps
\]
for every $f\in\F$ and every $|y|<\rho$. Taking the supremum over
$f\in\F$ proves the claim.
\end{proof}

\section{A converse through compactness}

The reverse implication requires more care. Translation control alone does not
control the vertical size of the functions. The compactness theorem in
asymptotic $L^p$ spaces shows that almost equiboundedness is the missing
hypothesis.

\begin{theorem}\label{thm:tc-to-ae}
Let $1\le p<\infty$. Let $E\subset\R^n$ be measurable with $m(E)<\infty$,
and let $\F$ be a family of measurable functions supported in $E$. Assume that
\begin{enumerate}
    \item $\F$ is almost equibounded;
    \item $\F$ satisfies the $p$-truncated translation condition.
\end{enumerate}
Then $\F$ is almost equicontinuous.
\end{theorem}

\begin{proof}
Since all functions in $\F$ are supported in the same finite-measure set $E$,
the tail condition in the asymptotic Kolmogorov--Riesz theorem is automatic.
Indeed, for every $\delta>0$ we can choose $R>0$ such that
$m(E\cap\{|x|>R\})<\delta$, and then
\[
    \sup_{f\in\F}
    \int_{|x|>R}\min(|f(x)|,1)^p\,dx
    \le m(E\cap\{|x|>R\})<\delta .
\]
Together with the assumed translation condition and almost equiboundedness,
Alves's Kolmogorov--Riesz theorem in asymptotic $L^p$ spaces implies that
$\F$ is relatively compact in the topology generated by $\phip$.

On a finite-measure space, convergence in this truncated topology agrees with
convergence in measure. Therefore the restrictions of the functions in $\F$ to
$E$ form a relatively compact family for convergence in measure. By the
classical compactness criterion for measurable functions, relative compactness
in measure is equivalent to almost equiboundedness together with almost
equicontinuity. Since almost equiboundedness is already assumed, the remaining
conclusion is that $\F$ is almost equicontinuous.
\end{proof}

\begin{corollary}\label{cor:equivalence}
For almost equibounded families supported in a fixed finite-measure set,
classical almost equicontinuity and the $p$-truncated translation condition are
equivalent.
\end{corollary}

\begin{proof}
The implication from almost equicontinuity to the translation condition is
Theorem~\ref{thm:ae-to-tc}. The converse is Theorem~\ref{thm:tc-to-ae}.
\end{proof}

\section{Examples and sharpness}

\begin{example}[A common exceptional set is too strong]\label{ex:common}
Let $n=1$ and
\[
    \F:=\{\mathbf{1}_{(t,t+1/2)}:0\le t\le 1\}.
\]
Every function in $\F$ is supported in $(0,3/2)$. For each $t$ and $y$,
\[
    \int_{\R}\min\bigl(
    |\mathbf{1}_{(t,t+1/2)}(x+y)-\mathbf{1}_{(t,t+1/2)}(x)|,1
    \bigr)^p\,dx
    \le 2|y|.
\]
Thus $\F$ satisfies the truncated translation condition.

The family is also almost equicontinuous in the sense of
Definition~\ref{def:ae}: for a fixed $t$, remove small neighborhoods of the
two jump points $t$ and $t+1/2$.

However, one cannot generally choose a single exceptional set that works for
all $t$. Indeed, if $B\subset\R$ has $m(B)<1$, then
$[0,1]\setminus B$ has positive measure. Hence for every $\rho>0$ there are
points $x<z$ in $[0,1]\setminus B$ with $z-x<\min(\rho,1/2)$. Taking
$t=(x+z)/2$, we have
\[
    \mathbf{1}_{(t,t+1/2)}(x)=0,
    \qquad
    \mathbf{1}_{(t,t+1/2)}(z)=1.
\]
So equicontinuity outside one common exceptional set fails. This is why the
classical formulation allows the exceptional set to depend on the function.
\end{example}

\begin{example}[Almost equiboundedness is independent]\label{ex:aeb}
Let $E=(0,1)$ and
\[
    \F:=\{k\mathbf{1}_E:k\in\mathbb{N}\}.
\]
The family satisfies the truncated translation condition, since
\[
    \int_{\R}\min(|k\mathbf{1}_E(x+y)-k\mathbf{1}_E(x)|,1)^p\,dx
    \le m(E\triangle(E-y))\to 0
\]
as $y\to 0$, uniformly in $k$. The family is also almost equicontinuous after
removing small neighborhoods of the endpoints of $E$.

Nevertheless, $\F$ is not almost equibounded: if $M>0$ and $k>M$, then
\[
    m(\{|k\mathbf{1}_E|>M\})=m(E)=1.
\]
Thus almost equiboundedness is a genuine additional hypothesis in compactness
results for asymptotic $L^p$ spaces.
\end{example}

\section{Connection with Alves's theorem}

Alves's Kolmogorov--Riesz theorem for asymptotic $L^p(\R^n)$ states that
relative compactness is characterized by three conditions: tail control,
truncated translation control, and almost equiboundedness. On families
supported in a fixed finite-measure set, the tail condition is automatic. The
results above identify the translation condition with classical almost
equicontinuity, once almost equiboundedness is included.

Thus, in the finite-support case, the compactness picture can be summarized as
\[
\boxed{
    \text{relative compactness}
    \quad\Longleftrightarrow\quad
    \text{almost equiboundedness}
    +
    \text{almost equicontinuity}
}
\]
where compactness is understood in the asymptotic $L^p$ topology, equivalently
in measure on the finite-measure support.
